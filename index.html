/* ===================================================== MODULO 1 CONFIGURACION GENERAL DEL SISTEMA Define nombres de hojas y utilidades del sistema ===================================================== */ const HOJA_REGISTRO = "REGISTRO"; const HOJA_USUARIOS = "usuarios"; /* ===================================================== MODULO 2 VALIDACION DE USUARIOS Verifica si un correo tiene acceso al sistema ===================================================== */ function usuarioActivo(email){ if(!email) return false; const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_USUARIOS); if(!sheet) return false; const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++){ const correo = String(data[i][0]).trim().toLowerCase(); const estado = String(data[i][1]).trim().toLowerCase(); if( correo === String(email).trim().toLowerCase() && estado === "si" ){ return true; } } return false; } /* ===================================================== MODULO 3 LECTURA DE REGISTROS (GET) - CORREGIDO Entrega datos como JSON estructurado ===================================================== */ function doGet() { try{ const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_REGISTRO); if(!sheet){ return respuestaJSON({ status:"error", mensaje:"Hoja REGISTRO no encontrada" }); } const data = sheet.getDataRange().getValues(); if(data.length === 0){ return respuestaJSON({ status:"ok", registros:[] }); } const headers = [ "fecha", "numero", "identMateria", "identDoc", "autoridad", "ejs", "hjs", "materia", "plazo", "procedencia", "clasificacion", "archivo", "llegsal", "observ" ]; const registros = data.slice(1).map(row => { let obj = {}; headers.forEach((h,i)=>{ obj[h] = row[i] || ""; }); return obj; }); return respuestaJSON({ status:"ok", registros:registros }); } catch(error){ return respuestaJSON({ status:"error", mensaje:error.toString() }); } } /* ===================================================== MODULO 4 REGISTRO DE DOCUMENTOS (POST) Recibe datos desde la aplicación web ===================================================== */ function doPost(e) { try{ const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(HOJA_REGISTRO); if(!sheet){ return respuestaJSON({ status:"error", mensaje:"Hoja REGISTRO no encontrada" }); } if(!e.postData || !e.postData.contents){ return respuestaJSON({ status:"error", mensaje:"No se recibieron datos" }); } const data = JSON.parse(e.postData.contents); sheet.appendRow([ data.fecha || "", data.numero || "", data.identMateria || "", data.identDoc || "", data.autoridad || "", data.ejs || "", data.hjs || "", data.materia || "", data.plazo || "", data.procedencia || "", data.clasificacion || "", data.archivo || "", data.llegsal || "", data.observ || "" ]); return respuestaJSON({ status:"ok", mensaje:"Documento registrado correctamente" }); } catch(error){ return respuestaJSON({ status:"error", mensaje:error.toString() }); } } /* ===================================================== MODULO 5 SOPORTE CORS Permite que la aplicación web pueda comunicarse con el Apps Script desde cualquier dominio ===================================================== */ function doOptions(){ return ContentService.createTextOutput(""); } /* ===================================================== MODULO 6 RESPUESTA JSON DEL SERVIDOR ===================================================== */ function respuestaJSON(data){ return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
